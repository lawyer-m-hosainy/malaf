```javascript
// src/main.tsx
Sentry.init({
```

الخطوة 2 — تصميم منظومة التنبيه (Alerting Strategy):
 
قاعدة التنبيه الذكي: لا تُنبّه على كل خطأ — نبّه على الأنماط.
 
| المُشغِّل (Trigger) | العتبة | القناة | الأولوية |
|---|---|---|---|
| Error Rate > 1% خلال 5 دقائق | 50 خطأ/5 دقائق | WhatsApp + Email | 🔴 حرج |
| Error Rate > 0.1% مستمرة | كل ساعة | Email | 🟠 عالية |
| Latency P95 > 2s لمسار حرج | 3 دقائق متتالية | Slack/Email | 🟠 عالية |
| Uptime انقطاع > 1 دقيقة | فوري | WhatsApp + Email | 🔴 حرج |
| Supabase Connection Pool > 80% | 5 دقائق | Email | 🟡 متوسطة |
| ETA API Error Rate > 5% | 10 دقائق | Email | 🟠 عالية |
| New Error Type (لم يحدث من قبل) | فوري | Email | 🟡 للمراجعة |
 
مُنبّهات Supabase Database (عبر pg_stat + Cron):
```sql
-- تشغيله كل 5 دقائق عبر Supabase Edge Function + Cron
SELECT
  (SELECT count(*) FROM pg_stat_activity WHERE state = 'active') AS active_connections,
  (SELECT count(*) FROM pg_stat_activity WHERE wait_event_type = 'Lock') AS blocking_queries,
  (SELECT pg_database_size('postgres') / 1024 / 1024) AS db_size_mb,
  (SELECT sum(n_dead_tup) FROM pg_stat_user_tables) AS total_dead_tuples;
-- أرسل تنبيهاً إذا: active_connections > 45، أو blocking_queries > 0
```

```typescript
// src/ai/prompt-sanitizer.ts
function sanitizeUserInput(input: string): string {
  const INJECTION_PATTERNS = [
    /ignore (previous|all|above) instructions?/gi,
    /you are now/gi,
    /new instructions?:/gi,
    /<\|system\|>/gi,
    /\[system\]/gi,
    /act as (an? )?(different|new|unrestricted)/gi,
  ];
 
  let sanitized = input;
  INJECTION_PATTERNS.forEach(pattern => {
    if (pattern.test(sanitized)) {
      // سجّل المحاولة في Audit Log
      logSecurityEvent('prompt_injection_attempt', { input, pattern: pattern.source });
      sanitized = sanitized.replace(pattern, '[محتوى محذوف لأسباب أمنية]');
    }
  });
 
  return sanitized;
}
 
// تُطبَّق على كل حقل يذهب للـ AI — ليس على مدخل المستخدم مباشرة فقط
function buildSafePrompt(template: string, data: DocumentData): string {
  const sanitizedData = Object.fromEntries(
    Object.entries(data).map(([k, v]) => [k, sanitizeUserInput(String(v))])
  );
  return fillTemplate(template, sanitizedData);
}
```

الخطوة 4 — بناء Fallback متعدد الطبقات:
```typescript
// src/ai/document-generator.ts
async function generateLegalDocument(type: DocumentType, data: DocumentData) {
  const FALLBACK_CHAIN = [
    { provider: 'gemini-flash', timeout: 10000 },  // الأسرع — للديمو
    { provider: 'claude-haiku', timeout: 15000 },  // الإنتاج الأساسي
    { provider: 'groq-llama3', timeout: 8000 },    // سريع وموثوق
    { provider: 'template-only', timeout: 100 },   // لا AI — قالب فقط
  ];
 
  for (const provider of FALLBACK_CHAIN) {
    try {
      const result = await Promise.race([
        generateWithProvider(provider.provider, type, data),
        new Promise((_, reject) =>
          setTimeout(() => reject(new Error('Timeout')), provider.timeout)
        )
      ]);
 
      // تحقق من الجودة قبل الإرجاع
      const quality = await checkDocumentQuality({ documentType: type, generatedText: result, sourceData: data });
      if (quality.safe_to_use) {
        trackAIUsage(provider.provider, type, quality.overall_score);
        return { document: result, provider: provider.provider, quality };
      }
      // الجودة منخفضة؟ جرب الخيار التالي
    } catch (error) {
      trackAIError(provider.provider, error);
      // جرّب الخيار التالي في السلسلة
    }
  }
 
  throw new Error('جميع نماذج AI غير متاحة — يُرجى المحاولة لاحقاً');
}
```

الخطوة 1 — مراجعة GitHub Actions Workflow الحالي وبناء CI كامل:
```yaml
# .github/workflows/ci.yml
name: CI — malaf.pro Quality Gate
 
on:
  pull_request:
    branches: [main, staging]
  push:
    branches: [main]
 
jobs:
  quality-gate:
    runs-on: ubuntu-latest
    timeout-minutes: 15
 
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: '20', cache: 'npm' }
 
      - name: Install Dependencies
        run: npm ci
 
      - name: Type Check
        run: npm run type-check    # tsc --noEmit
 
      - name: Lint
        run: npm run lint          # eslint src --max-warnings 0
 
      - name: Unit & Integration Tests
        run: npm run test:coverage
        env:
          SUPABASE_URL: ${{ secrets.STAGING_SUPABASE_URL }}
          SUPABASE_ANON_KEY: ${{ secrets.STAGING_SUPABASE_ANON_KEY }}
 
      - name: Check Coverage Thresholds
        run: |
          COVERAGE=$(cat coverage/coverage-summary.json | jq '.total.lines.pct')
          if (( $(echo "$COVERAGE < 80" | bc -l) )); then
            echo "❌ Code coverage $COVERAGE% is below 80% threshold"
            exit 1
          fi
 
      - name: Security Audit
        run: npm audit --audit-level=high
 
      - name: Build
        run: npm run build
        env:
          VITE_SUPABASE_URL: ${{ secrets.STAGING_SUPABASE_URL }}
          VITE_SUPABASE_ANON_KEY: ${{ secrets.STAGING_SUPABASE_ANON_KEY }}
          VITE_APP_VERSION: ${{ github.sha }}
 
      - name: Bundle Size Check
        run: |
          BUNDLE_SIZE=$(du -sk dist/assets/*.js | awk '{sum+=$1} END {print sum}')
          echo "Bundle size: ${BUNDLE_SIZE}KB"
          if [ $BUNDLE_SIZE -gt 500 ]; then
            echo "⚠️ Bundle size ${BUNDLE_SIZE}KB exceeds 500KB limit"
            exit 1
          fi
 
      - name: Upload Coverage to Codecov
        uses: codecov/codecov-action@v3
        with:
          token: ${{ secrets.CODECOV_TOKEN }}
 
  e2e-tests:
    needs: quality-gate
    runs-on: ubuntu-latest
    timeout-minutes: 20
 
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: '20', cache: 'npm' }
      - run: npm ci
      - run: npx playwright install --with-deps chromium
 
      - name: Run Smoke Tests (Critical Paths Only in CI)
        run: npx playwright test --project=chromium tests/e2e/smoke/
        env:
          BASE_URL: ${{ secrets.STAGING_URL }}
          TEST_USER_EMAIL: ${{ secrets.STAGING_TEST_EMAIL }}
          TEST_USER_PASSWORD: ${{ secrets.STAGING_TEST_PASSWORD }}
```

الخطوة 3 — آلية Rollback الفوري (≤ 10 دقائق):
 
أ. Rollback على Render.com:
```
الطريقة الأسرع (3 دقائق):
Render Dashboard → malaf-pro Service → Deploys → اختر آخر Deploy ناجح → "Redeploy"
 
الطريقة الآمنة عبر Git (5-10 دقائق):
git revert <commit-hash> --no-commit
git commit -m "revert: إصلاح عاجل — التراجع عن [وصف]"
git push origin main
# Render سيُشغِّل Deploy تلقائياً
```
 
ب. Rollback قاعدة البيانات (الأخطر):
```sql
-- قبل أي Migration في الإنتاج، نفّذ دائماً:
-- 1. تحقق من وجود Down Migration
-- 2. خذ Snapshot يدوي من Supabase Dashboard
-- 3. نفّذ Migration
 
-- إذا احتجت التراجع:
-- Supabase → Database → Migrations → اختر Migration → "Rollback"
-- أو نفّذ يدوياً: supabase db reset --db-url $PROD_DB_URL (خطير جداً)
 
-- الحل الأكثر أماناً: Feature Flags لكل تغيير DB
```
